40T couetteVelocity(plint iY, T shearRateLU, T zeroCoordinate)
41{
42 return shearRateLU*(iY - zeroCoordinate);
43}
44template <typename T>
45CouetteDensityAndVelocity<T>::CouetteDensityAndVelocity(T shearRateLU_, T zeroCoordinate_)
46: shearRateLU(shearRateLU_), zeroCoordinate(zeroCoordinate_)
47{ }

49template <typename T>
50void CouetteDensityAndVelocity<T>::operator()(plint iX, plint iY, plint iZ, T &rho, hemo::Array<T,3>& u) const {
51 rho = (T)1;
52 u[0] = couetteVelocity(iY, shearRateLU, zeroCoordinate);
53 u[1] = T();
54 u[2] = T();
55}
